// Display an info toast with no title
toastr.options.closeButton = true;
toastr.options.progressBar = true;
toastr.options.onShown = function() { 
	// seta a data de envio da notificacao
	$.ajax({
		url: '<?= Lb_URL::url(array('controller'=>'Notificacoes', 'action'=>'alterar_data_envio')) ?>',
		type: 'GET',
		data: {
			id_notificacao: notificacao.id_notificacao
		},
	});
}
toastr.options.onHidden = function() { console.log('goodbye'); }
toastr.options.onclick = function() { 
	window.open(location.origin + '/Notificacoes/visualizar_notificacao/?id_notificacao=' + notificacao.id_notificacao, '_blank');
}
toastr.options.onCloseClick = function() { console.log('close button clicked'); }
toastr.options.timeOut = 15000; // How long the toast will display without user interaction
toastr.options.extendedTimeOut = 5000; // How long the toast will display after a user hovers over it
toastr.info(notificacao.descricao, notificacao.titulo)
