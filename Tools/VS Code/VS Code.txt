# atalhos

# ocultar todos os blocos de código
ctrl + k + 0

# exibir todos os blocos de cógido
ctrl + k + j

# copiar linha 
shift + alt + seta para cima ou baixo 

# mover linha 
alt + seta para cima ou baixo 

# apagar linha 
ctrl + shift + k 

# multipla seleção (mesmo texto)
ctrl + d 

# multipla seleção (texto diferente)
alt + click 

# ativar zen mode 
(ctrl + k) e z

# desativar zen mode 
2x esc 

# abrir intelisense 
ctrl + espaço

# ver todos os atalhos 
ctrl + k e ctrl + s 

# abrir extensões 
ctrl + shift + x
